# VideoCustomization

## Example Usage

```typescript
import { VideoCustomization } from "@wistia/wistia-api-client/models";

let value: VideoCustomization = {
  plugin: {
    postRollV1: {
      style: {
        backgroundColor: "#303030",
      },
      ctaType: "text",
    },
  },
  videoFoam: true,
};
```
